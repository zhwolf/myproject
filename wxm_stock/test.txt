	function ConvertCurrency(redirectPath, amt, from, to)
	{
		if(calculate()) {
			var amtCtrl = document.getElementById(amt);
			var amtCtrlValue = amtCtrl.value;
			var invalidChars = amtCtrlValue.match(/\,/g);
			if (invalidChars != null) {
				var updatedAmtCtrl = amtCtrlValue.replace(/,/g, "");
			} else {
				var updatedAmtCtrl = amtCtrlValue;
			}
			var fromCtrl = document.getElementById(from);
			var toCtrl = document.getElementById(to);
			redirectPath = redirectPath.replace("[amt]", updatedAmtCtrl);

function validateAmount() {
	var valid = false;
	//var srcAmtElem = document.getElementById("srcAmt");
	var srcAmtElem = document.getElementById("CurrenciesConversion_ConversionAmount");
	var amount = srcAmtElem.value;
	if (amount != null && amount.length>0) {
		var dotCount = 0;
		var newAmount = '';
		var amountChars = amount.match(/\d|\.|\,/g);
		if (amountChars != null) {
			for (var i=0; i < amountChars.length; i++) {
				if (amountChars[i] == '.') dotCount++;
				if ( dotCount > 1 ) break;
				newAmount = newAmount + amountChars[i];
			}
			if (newAmount == amount && dotCount<2) { 
				valid = true;
			}
		}
	}
	return valid;
}
